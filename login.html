import { auth, db } from "./core/firebase-config.js"; import { signInWithEmailAndPassword } from "https://www.gstatic.com/firebasejs/10.12.2/firebase-auth.js"; import { doc, getDoc, setDoc, serverTimestamp } from "https://www.gstatic.com/firebasejs/10.12.2/firebase-firestore.js"; const loginForm = document.getElementById("loginForm"); const msgError = document.getElementById("msgError"); // El cuadro rojo de tus fotos /** * PROTOCOLO 1: INICIO DE SESIÓN MANUAL (SuperAdmin / Staff) */ loginForm.onsubmit = async (e) => { e.preventDefault(); const email = document.getElementById("email").value.trim(); const pass = document.getElementById("password").value.trim(); try { const userCredential = await signInWithEmailAndPassword(auth, email, pass); const uid = userCredential.user.uid; // 🛡️ BÚSQUEDA EN COLECCIÓN RAÍZ (La clave del éxito) const userRef = doc(db, "usuarios", uid); const userSnap = await getDoc(userRef); if (userSnap.exists()) { const data = userSnap.data(); // Inyectamos todo al LocalStorage para el Core localStorage.setItem("nexus_uid", uid); localStorage.setItem("nexus_empresaId", data.empresaId); localStorage.setItem("nexus_rol", data.rol || "TECNICO"); window.location.href = "app/index.html"; } else { throw new Error("NODO NO RECONOCIDO EN LA RED GLOBAL"); } } catch (error) { msgError.innerText = `FALLO EN SECUENCIA: ${error.message.toUpperCase()}`; msgError.classList.remove("hidden"); } }; /** * PROTOCOLO 2: ACCESO FREEMIUM / DEMO AUTOMÁTICO * Este es el que me pediste para los 7 días sin fricción. */ window.iniciarDemo = async () => { const emailDemo = prompt("Ingrese su Email para activar los 7 días de prueba:"); if (!emailDemo) return; try { // 1. Generamos un ID de Empresa temporal const tempEmpresaId = "DEMO_" + Math.random().toString(36).substring(7).toUpperCase(); // 2. Aquí podrías usar createUserWithEmailAndPassword o un login anónimo // Para simplificar y que entren YA, simulamos el registro en la raíz: const demoUid = "TEMP_" + Date.now(); // Creamos el perfil de prueba en la raíz 'usuarios' await setDoc(doc(db, "usuarios", demoUid), { email: emailDemo, empresaId: tempEmpresaId, rol: "ADMIN", plan: "FREE_TRIAL", expira: new Date(Date.now() + 7 * 24 * 60 * 60 * 1000), // 7 días exactos createdAt: serverTimestamp() }); // Guardamos y saltamos al Dashboard localStorage.setItem("nexus_uid", demoUid); localStorage.setItem("nexus_empresaId", tempEmpresaId); localStorage.setItem("nexus_rol", "ADMIN"); alert("¡Nodo Demo Activado! Bienvenido a la red Nexus-X."); window.location.href = "app/index.html"; } catch (error) { alert("Error al sincronizar Demo: " + error.message); } };
